# 示例：自动生成规划分析图
import ezdxf

def generate_land_use_plan(land_data):
    """根据用地数据自动生成总平面图"""
    doc = ezdxf.new('R2010')
    msp = doc.modelspace()
    
    # 绘制用地红线
    for plot in land_data:
        msp.add_lwpolyline(plot['coords'], close=True)
        # 填充不同颜色表示不同功能
        # 添加标注
        msp.add_text(plot['use'], height=2.5).set_pos(plot['center'])
    
    doc.saveas('plan.dxf')
```

import bpy

def generate_building_massing(floor_area, height, style_params):
    """根据参数自动生成建筑体块"""
    # 清除场景
    bpy.ops.object.select_all(action='SELECT')
    bpy.ops.object.delete()
    
    # 创建建筑基底
    bpy.ops.mesh.primitive_cube_add(size=1)
    building = bpy.context.active_object
    building.scale = (floor_area**0.5, floor_area**0.5, height/2)
    
    # 应用风格参数（新中式/现代/ etc）
    apply_style(building, style_params)
    
    return building
```
